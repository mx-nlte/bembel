31  inline void init_VTKSurfaceExport(const Geometry& geo, int M) {
32  msh.init_ClusterTree(geo, M);
33  points = msh.get_element_tree().generatePointList().transpose();
34  cells = msh.get_element_tree().generateElementList().transpose();
35  normals = Eigen::MatrixXd(cells.rows(), 3);
36  patch_number = Eigen::VectorXi(cells.rows());
37 
38  for (auto e = msh.get_element_tree().cpbegin();
39  e != msh.get_element_tree().cpend(); ++e) {
40  normals.row(e->id_) = msh.get_geometry()[e->patch_]
41  .evalNormal(e->referenceMidpoint())
42  .normalized()
43  .transpose();
44  }
45  for (auto e = msh.get_element_tree().cpbegin();
46  e != msh.get_element_tree().cpend(); ++e) {
47  patch_number(e->id_) = e->patch_;
48  }
49  return;
50  }
51  // One can add data to visualize via the addDataSet methods. They accept a
52  // std::function object of different types, and generate the data needed for
53  // the vtk file. Allowed formats are:
54  // std::function<double(int, Eigen::Vector2d)>& fun)
55  // std::function<Eigen::Vector3d(int, Eigen::Vector2d)>
56  // std::function<double(Eigen::Vector3d)>
57  // std::function<Eigen::Vector3d(Eigen::Vector3d)>
58  inline void addDataSet(
59  const std::string& name,
60  std::function<double(int, const Eigen::Vector2d&)> fun) {
61  Eigen::MatrixXd data(cells.rows(), 1);
62  for (auto e = msh.get_element_tree().cpbegin();
63  e != msh.get_element_tree().cpend(); ++e) {
64  data(e->id_) = fun(e->patch_, e->referenceMidpoint());
65  }
66  addDataSet_(name, data);
67  return;
68  }
69  inline void addDataSet(
70  const std::string& name,
71  std::function<Eigen::Vector3d(int, const Eigen::Vector2d&)> fun) {
72  Eigen::MatrixXd data(cells.rows(), 3);
73  int k = 0;
74  for (auto e = msh.get_element_tree().cpbegin();
75  e != msh.get_element_tree().cpend(); ++e) {
76  data.row(e->id_) = fun(e->patch_, e->referenceMidpoint()).transpose();
77  }
78  addDataSet_(name, data);
79  return;

106  inline void writeToFile(const std::string& filename) {
107  std::ofstream output;
108  output.open(filename);
109  output << "<VTKFile type=\"PolyData\" version=\"0.1\" "
110  "byte_order=\"LittleEndian\">\n"
111  "<PolyData>\n"
112  "<Piece NumberOfPoints=\" "
113  << points.rows()
114  << "\" NumberOfVerts=\"0\" NumberOfLines=\"0\" NumberOfStrips=\"0\" "
115  "NumberOfPolys=\""
116  << cells.rows()
117  << "\">\n"
118  "<Points>\n"
119  "<DataArray type=\"Float32\" NumberOfComponents=\"3\" "
120  "format=\"ascii\">\n";
121  output << points;
122  output << "</DataArray>\n</Points>";
123  output << "<CellData Scalars=\"patch_number\" Normals=\"cell_normals\">\n"
124  "<DataArray type=\"Int32\" Name=\"patch_number\" "
125  "format=\"ascii\">\n";
126  output << patch_number;
127  output << "</DataArray>\n<DataArray type=\"Float32\" Name=\"cell_normals\" "
128  "NumberOfComponents=\"3\" format=\"ascii\">";
129  output << normals;
130  output << "</DataArray>\n";
131  for (auto data : additionalData) {
132  output << data;
133  }
134  output << "</CellData>\n"
135  "<Polys>\n"
136  "<DataArray type=\"Int32\" Name=\"connectivity\" "
137  "format=\"ascii\">\n";
138  output << cells;
139  output << "</DataArray>\n"
140  "<DataArray type=\"Int32\" Name=\"offsets\" "
141  "format=\"ascii\">\n";
142  for (int i = 0; i < cells.rows(); ++i) {
143  output << (i + 1) * 4;
144  output << "\n";
145  }
146  output << "</DataArray>\n"
147  "</Polys>\n"
148  "</Piece>\n"
149  "</PolyData>\n"
150  "</VTKFile>";
151  output.close();
152  return;
153  }

160  private:
161  // This routine turns the data of the above DataSet-routines into a string and
162  // stores it.
163  inline void addDataSet_(const std::string& name, const Eigen::MatrixXd& mat) {
164  assert(mat.cols() == 1 || mat.cols() == 3);
165  assert(mat.rows() == cells.rows());
166 
167  const int cols = mat.cols();
168  std::string data_ascii = "<DataArray type=\"Float32\" Name=\"" + name +
169  "\" NumberOfComponents=\"" +
170  std::to_string(mat.cols()) +
171  "\" format=\"ascii\">\n";
172  std::ostringstream out;
173  out.precision(6);
174  for (int i = 0; i < mat.rows(); ++i) {
175  for (int j = 0; j < cols; ++j) {
176  out << std::scientific << mat(i, j);
177  data_ascii.append(std::move(out).str() + " ");
178  out.str("");
179  out.clear();
180  }
181  data_ascii.append("\n");
182  }
183  data_ascii.append("</DataArray>\n");
184  additionalData.push_back(data_ascii);
185  }
